2. Create `packages/api/src/lib/logger-config.ts` with a single exported function `buildLoggerOptions`:

```typescript
import type { FastifyServerOptions } from 'fastify';

const PINO_LEVEL_TO_GCP_SEVERITY: Record<string, string> = {
  trace: 'DEBUG',
  debug: 'DEBUG',
  info: 'INFO',
  warn: 'WARNING',
  error: 'ERROR',
  fatal: 'CRITICAL',
};

interface LoggerEnv {
  LOG_LEVEL?: string;
  LOG_PRETTY?: string;
  LOG_FORMAT?: string;
  NODE_ENV?: string;
}

export function buildLoggerOptions(
  env: LoggerEnv,
): NonNullable<FastifyServerOptions['logger']> {
  const isProduction = env.NODE_ENV === 'production';
  const level = env.LOG_LEVEL || (isProduction ? 'warn' : 'debug');
  const pretty = env.LOG_PRETTY === 'true';
  const gcpFormat = env.LOG_FORMAT === 'gcp' || isProduction;

  // Base options
  const opts: Record<string, unknown> = { level };

  if (gcpFormat && !pretty) {
    // GCP Cloud Logging format (per D-01 spirit — inline config replaces
    // pino-cloud-logging which is incompatible with pino 9.x)
    opts.messageKey = 'message';
    opts.formatters = {
      level(label: string, number: number) {
        return {
          severity: PINO_LEVEL_TO_GCP_SEVERITY[label] ?? 'DEFAULT',
          level: number,
        };
      },
    };
    // GCP Error Reporting: map err.stack to stack_trace field (per D-09)
    opts.serializers = {
      err(err: Error) {
        return {
          type: err.constructor?.name ?? 'Error',
          message: err.message,
          stack_trace: err.stack,
        };
      },
    };
  }

  if (pretty) {
    // pino-pretty transport — opt-in via LOG_PRETTY=true (per D-02)
    opts.transport = {
      target: 'pino-pretty',
      options: {
        translateTime: 'HH:MM:ss Z',
        ignore: 'pid,hostname',
      },
    };
  }

  return opts as NonNullable<FastifyServerOptions['logger']>;
}
```

Key design points:
- Reads from process.env at call time (Pitfall 5: @fastify/env loads too late for logger init)
- Uses `||` not `??` for LOG_LEVEL so empty string falls through to default
- GCP format is NOT applied when pretty is active (pretty transport overrides)
- pino is NOT added as a direct dependency (anti-pattern from research)
- err serializer maps `err.stack` to `stack_trace` for GCP Error Reporting auto-detection (D-09)

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| client -> API (HTTP) | X-Request-ID header accepted from upstream; untrusted input used for log correlation |
| env vars -> logger | LOG_LEVEL, LOG_PRETTY, LOG_FORMAT control logger behavior; trusted (operator-set) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-01 | Spoofing | requestIdHeader (X-Request-ID) | accept | IDs are for log correlation only, not authorization. Upstream GCP LB sets legitimate IDs. Spoofed IDs cannot escalate privilege -- they only affect log readability. Per D-05 and research security analysis. |
| T-05-02 | Information Disclosure | Error serializer (stack_trace) | mitigate | Stack traces are emitted to stdout logs only, not in HTTP responses. Fastify's error handler returns sanitized error responses (status + message). The err serializer in logger-config.ts is pino-internal -- it shapes log output, not API responses. |
| T-05-03 | Tampering | Log injection via request headers | mitigate | Pino serializes all values as JSON with automatic escaping. Special characters in X-Request-ID or other logged fields cannot break JSON structure or inject newlines. No additional sanitization needed. |
</threat_model>
